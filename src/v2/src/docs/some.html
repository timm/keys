    function Some:add1(x)
  if #self._all < self.most then 
    self.sorted = false
    self._all[1 + #self._all] = x
  elseif rand() < #self._all/self.n then 
    self.sorted = false 
    self._all[ math.floor(1+ rand()*#self._all) ] = x end end
