    randi(?lo:num, ?hi:num)

    Return a float between lo and hi (default 0..1).

    local function rand(lo,hi,     mult,mod)
  lo,hi = lo or 0, hi or 1
  mult, mod = 16807, 2147483647
  Seed = (mult * Seed) % mod 
  return lo + (hi-lo) * Seed / mod end
